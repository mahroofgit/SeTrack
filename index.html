function renderGroupedList(items, container, isTrash) { const query = document.getElementById('search-input').value.toLowerCase(); const filtered = items.filter(i => i.name.toLowerCase().includes(query)); const groups = filtered.reduce((acc, item) => { const d = item.date || "Legacy"; if (!acc[d]) acc[d] = []; acc[d].push(item); return acc; }, {}); Object.keys(groups).sort((a,b) => b.localeCompare(a)).forEach(date => { const header = document.createElement('div'); header.className = "flex justify-center my-4"; header.innerHTML = `${date}`; container.appendChild(header); groups[date].sort((a,b) => a.name.localeCompare(b.name)).forEach(item => { const card = document.createElement('div'); if (isTrash) { card.className = "p-4 bg-slate-200/50 border border-slate-300 rounded-2xl opacity-80 mb-2"; card.innerHTML = `
${item.name}
`; } else { const roi = item.sold ? ((item.profit / item.totalPaid) * 100).toFixed(0) : 0; let profitClass = item.profit < 0 ? "text-red-600" : (roi >= 100 ? "high-roi" : "text-green-600"); card.className = `p-4 rounded-2xl flex justify-between items-center border mb-2 ${item.sold || item.archived ? 'bg-slate-100 border-slate-200' : 'bg-white border-slate-200 shadow-sm'}`; let mainAction = item.sold ? `

${item.profit >= 0 ? '+' : ''}$${item.profit.toFixed(2)}

${roi}% ROI

` : (item.archived ? `` : ``); card.innerHTML = `

${item.name}

Bid: $${item.listedPrice.toFixed(2)} | Cost: $${item.totalPaid.toFixed(2)} ${item.sold ? ` | Sale: $${item.sell.toFixed(2)}` : ''} ${item.archived ? ` | [ARCHIVED]` : ''}
${mainAction} ${!item.sold ? `
${!item.archived ? ` ` : ''} ${item.archived ? ` ` : ''}
` : ''}
`; } container.appendChild(card); }); }
